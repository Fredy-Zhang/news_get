                                                <p class='t-body-text'>An Israeli district court has ruled to allow a young settler, suspected of killing a Palestinian woman after throwing stones at her car, to return to his home, despite objections by the Shin Bet security service and state prosecutors, who are concerned that he will be further radicalized.
                                                    </p>
                
                
                                    
                
                                                                                                                                                                                                                                                                                                                                                    
                                                                                                                                                        <p class='t-body-text'>The adolescent, who cannot be publicly named under Israeli law, was until now living under house arrest with his grandparents within the Green Line, but is originally from the West Bank settlement of Kokhav Hashahar, east of Ramallah. The conditions of his house arrest will continue there, tagged with an electronic tracking device and under the supervision of his family.
                                                    </p>
                
                
                                    
                
                                                                                                                                                                                                                                                                                                                                                    
                                                                                                                                                        <p class='t-body-text'>The minor was arrested in October 2018 at <a href="https://www.haaretz.com/israel-news/.premium-yeshiva-where-suspects-in-murder-of-palestinian-woman-study-seen-as-strange-bird-1.6822845">a religious school in the settlement of Rehelim</a>, along with other students who were suspected of throwing rocks at the vehicle of a Palestinian family. <a href="https://www.haaretz.com/israel-news/shin-bet-jewish-minors-arrested-over-murder-of-palestinian-woman-1.6810146">According to the indictment</a>, the accused and other students at the Pri Haaretz yeshiva were standing along Highway 60 between the Rechelim and Tapuach junctions, south of Nablus, &ldquo;when the youth lifted a rock weighing almost two kilograms with the intent of using it to harm passengers in an Arab vehicle, driven by a racist ideology and hostility towards Arabs.&rdquo;
                                                    </p>

	var articlePageContentGroup =  {
	   "Author"  :"Hagar Shezaf,Netael Bandel",
	   "categType"	  :"Israel News" ,
	   "openType"	: !!location.href.match('premium') ? 'premium' : 'not premium',
	   "layout"	:getArticleLayout()
	}

	$(document).ready(function(){
		try{
			var articleMode = getArticleMode(UserUtil.getLoginInfo());
			GaTm.articlePage.init(articlePage,UserUtil.getLoginInfo());
			GaTm.articlePage.initContentGroup(articlePageContentGroup,UserUtil.getLoginInfo());
			//GaTm.eventManager.fireArticlePageEvent();
			ga('set', 'dimension1', GaTm.articlePage.getusrType()); //User type
			ga('set', 'dimension2', articlePageContentGroup.Author); //Author Name - Author
			//ga('set', 'dimension3', articlePageContentGroup.openType); //Article Mode - openType
			ga('set', 'dimension3', "" +articleMode); //Article Mode - openType
			ga('set', 'dimension4', articlePageContentGroup.layout); //Page Type - layout
			//ga('send', 'pageview');

			var fullReadFireEvent,halfReadFiredEvent = false;
			var lastPTag = $(".article-entry p:last");
			$(window).on('smartscroll', function(event) {
				if (!halfReadFiredEvent && $(window).scrollTop()>200)
				{
					console.log("Ga Half Read")
					GaTm.eventManager.fireHalfReadEvent();
					halfReadFiredEvent = true;
				}
				if (lastPTag.size()>0){
					if (!fullReadFireEvent && $(window).scrollTop() > lastPTag.offset().top){
						console.log("Ga Full Read")
						GaTm.eventManager.fireFullReadEvent();
						fullReadFireEvent=true;
					}
				}
			}).trigger('smartresize');
		}catch (err){console.error(err, err.stack);}
	});

	function getArticleLayout(){
		var contentType = "regularArticle";
		if(contentType.length > 0){
			return contentType;
		}else{
			var contentTypeObj = {
				"com.polobase.Tag": "Tag Page",
				"com.polobase.Author": "Author Page"
			};
			var contentType = "com.polobase.StandardArticle";
			if(contentType.length > 0){
				var articleLayout = contentTypeObj[contentType];
				if(articleLayout.length > 0){
					return articleLayout;
				}
			}
		}
		return "";
	}

	function getArticleMode(userLogginInfo){
		var articleModeData = {
			premiumClosed:"closed premium",
			premiumOpen:"opened premium",
			notPremium:"not premium",
			notPremiumClosed:"not premium closed",
			superPremiumClosed:"opened  superpremium",
			superPremiumOpen:"closed superpremium"
		};

		var paying = userLogginInfo.userType === 'paying';
		var premium = false;
		var superPremium = false;
		if(document.querySelector('[data-statutilispremiumcontent]') && document.querySelector('[data-statutilisSuperContent]')){
			premium = document.querySelector('[data-statutilispremiumcontent]').dataset["statutilispremiumcontent"];
			superPremium = document.querySelector('[data-statutilisSuperContent]').dataset["statutilisSuperContent"]
		}

		if(paying){
			if(superPremium && superPremium == "true"){
				return articleModeData.superPremiumOpen;
			}else if(premium && premium == "true"){
				return articleModeData.premiumOpen;
			}else{
				return articleModeData.notPremium;
			}
		}else{
									var articleBlocked = false;

			if(superPremium && superPremium == "true"){
				if(articleBlocked){
					return articleModeData.superPremiumClosed;
				}else{
					return articleModeData.superPremiumOpen;
				}
			}else if(premium && premium == "true"){
				if(articleBlocked){
					return articleModeData.premiumClosed;
				}else{
					return articleModeData.premiumOpen;
				}
			}else{
				if(articleBlocked){
					return articleModeData.notPremiumClosed;
				}else{
					return articleModeData.notPremium;
				}
			}
		}
	}
